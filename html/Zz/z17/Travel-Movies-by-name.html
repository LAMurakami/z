SQL Statement
SELECT
 SUBSTR(a.name,1,41) AS Name,
 b.path,
 a.date, a.time,
 b.size / 1073741824 AS 'Size (G)',
 count(a.name) AS Count
FROM movie_ls_l AS a
LEFT JOIN movie_du_sb AS b
  ON (a.name = b.name
  AND a.system = b.system
  AND a.path = b.path)
WHERE a.system = 'Toshiba-2TB'
   OR a.system = 'WD-2TB'
   OR a.system = 'USB-921GB-I'
   OR a.system = 'HD-300GB'
GROUP BY a.name, 'Size (G)'
HAVING b.path LIKE '%DVD'
ORDER BY a.name;
