SELECT
 a.name,
 a.size / 1048576 AS 'size (M)',
 a.path,
 a.date, a.time
FROM movie_ls_l AS a
LEFT JOIN movie_du_sb AS b
  ON (a.name = b.name
  AND a.system = b.system
  AND a.path = b.path)
WHERE a.path IN ('/SD-Card-03/3gp')
ORDER BY a.date DESC
