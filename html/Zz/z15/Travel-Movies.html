SELECT a.name,
 IF (b.size IS NULL,a.size / 1073741824,b.size / 1073741824) AS 'Size (G)',
 IF (b.size IS NULL,'f','d') AS Type,             # LEFT OUTER JOIN is file
 date, time, a.path
FROM movie_ls_l AS a
LEFT OUTER JOIN movie_du_sb AS b                  # INNER JOIN is directory
  ON (a.name = b.name
  AND a.system = b.system
  AND a.path = b.path)
WHERE a.system = "WD-2TB"   # Search clause
   OR a.system = "USB-921GB-I"   # Search clause
ORDER BY a.name, a.path;
